Maximum Depth of Binary Tree


public int maxDepth(TreeNode root){
TreeNode current=root;
if(current==null)return 0;

if(current.left==null&&current.right==null)return 1;

if(current.right==null) return maxDepth(current.left)+1;

if(current.left==null) return maxDepth(current.right)+1;

return Math.max(maxDepth(current.left)+1,maxDepth(current.right)+1);
}

Binary Search


/**
Starts at middle of data set, checking each iteration if the goal is greater than or less than the chosen data point
Precondition: Data must be sorted
Time Complexity: O(log N), could be more if not sorted
**/

public int search(int[]arr, int target){
int low=0;
int high=arr.length-1;
int mid=low+(high-low)/2;

while(low<high){
mid=low+(high-low)/2;
if(arr[mid]==target) return mid;
else if(arr[mid]>target){
high=mid-1;
}
else{
low=mid+1;
}
}

return -1;
}

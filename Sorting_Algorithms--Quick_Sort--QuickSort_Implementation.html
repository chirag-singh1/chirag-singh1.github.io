public static int partition(int[]arr, int min,int max){
int pivot=arr[max];
int ind=min;

for(int i=min;i<max;i++){
if(arr[i]<pivot){
int temp=arr[ind];
arr[ind]=arr[i];
arr[i]=temp;
ind++;
}
}

arr[max]=arr[ind];
arr[ind]=pivot;

return ind;
}
